thay bằng
#include <windows.h>
#include <iostream>

// Hàm tự định nghĩa gotoXY
void gotoXY(int x, int y) {
    COORD coord;
    coord.X = x;
    coord.Y = y;
    SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), coord);
}

int main() {
    system("cls"); // Xóa màn hình
    gotoXY(10, 5); // Di chuyển con trỏ đến tọa độ (10, 5)
    std::cout << "Hello at (10, 5)" << std::endl;
    return 0;
}
}
